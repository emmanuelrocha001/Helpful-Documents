#----------------------------------------------------------------------
Compiling on linux terminal
#----------------------------------------------------------------------

#----------------------------------------------------------------------
CS:

  (1)mcs program.cs //compile
  (2)mono program.exe //run
  (3)wc program.cs  //stats

#----------------------------------------------------------------------
PYTHON:

  //compile and run program
  (1) python3 program.py //compile and run
  (2) wc program.py //stats

#----------------------------------------------------------------------
JAVA:

  (1)javac program.java //compile
  (2)java program //run

#----------------------------------------------------------------------
C:

  (1)gcc -std=c++11 program.cpp -o program // compile
  (2)./program  //run

#----------------------------------------------------------------------
C:

  (1)gcc program.c -o program // compile
  (2)./program  //run

#----------------------------------------------------------------------
ARMS ASSEMBLY(using gcc(on an Arm Linux-like Platform)):

  (1)gcc -o program program.s  // compile
  (2)./program  //run
